 
#include <iostream>
#include<cmath>

using namespace std;

int main() {
    int n;
    cin >> n;
    int arr[n];
    for (int i = 0; i < n; i++) {
        cin >> arr[i];
    }
    int cnt = 0, mx = arr[0], mn = arr[0];
    for (int i = 1; i < n; ++i) {
        if (arr[i] > mx || arr[i] < mn)++cnt;
        mx = max(mx, arr[i]);
        mn = min(mn, arr[i]);
    }
    cout << cnt;
}
